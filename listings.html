/* ════════════════════════════════════════════════════ EXIT INTENT — "did you find what you were looking for?" 'block' : 'none'; }); } function exitClose() { document.getElementById('exit-overlay').classList.remove('open'); } function exitCloseAndGo() { exitClose(); if (exitPendingHref) window.location.href = exitPendingHref; } // Desktop: mouse moving up toward the tab/address bar document.addEventListener('mouseout', (e) => { if (e.clientY <= 0) exitOpenModal(null); }); // Any click that navigates away from this page (nav, logo, footer, mobile nav) document.addEventListener('click', (e) => { if (exitAlreadyShown()) return; e.preventDefault(); exitOpenModal(href); }); function exitAnswerYes() { if (localStorage.getItem(EXIT_REVIEW_DONE_KEY)) { exitCloseAndGo(); // already reviewed before, anywhere on the site — just let them go return; } exitStarRating = 0; document.getElementById('exit-review-nav').value = ''; document.getElementById('exit-review-improve').value = ''; document.getElementById('exit-review-name').value = ''; document.getElementById('exit-review-anon').checked = false; exitRenderStars(); exitShowPanel('review'); } function exitRenderStars() { const el = document.getElementById('exit-stars'); el.innerHTML = [1,2,3,4,5].map(n => `` ).join(''); } function exitSetStar(n) { exitStarRating = n; const checked = document.getElementById('exit-review-anon').checked; if (checked) { nameInput.dataset.saved = nameInput.value; nameInput.value = ''; nameInput.disabled = true; } else { nameInput.disabled = false; nameInput.value = nameInput.dataset.saved || ''; } } function exitSkipReview() { exitShowPanel('thanks'); document.getElementById('exit-thanks-text').textContent = 'No worries — thanks for visiting!'; const reviewerName = isAnon ? null : (typedName || null); const parts = []; if (nav) parts.push(`Navigation/finding things: ${nav}`); if (improve) parts.push(`Suggested improvement: ${improve}`); const comment = parts.join(' | ') || null; const btn = document.getElementById('exit-review-submit'); btn.disabled = true; btn.textContent = 'Submitting…'; const { error } = await db.from('reviews').insert({ review_type: 'platform', reviewer_role: 'buyer', reviewer_name: reviewerName, is_anonymous: isAnon, rating: exitStarRating, comment, }); btn.disabled = false; btn.textContent = 'Submit & continue'; if (error) { alert('Something went wrong, please try again: ' + error.message); return; } localStorage.setItem(EXIT_REVIEW_DONE_KEY, '1'); exitShowPanel('thanks'); document.getElementById('exit-thanks-text').textContent = 'Thank you for your feedback! 🎉'; setTimeout(exitCloseAndGo, 1400); } function exitAnswerNo() { // Only show the lightweight follow-up form some of the time — not every visit const showForm = Math.random() < 0.4; if (showForm) { document.getElementById('exit-lite-name').value = ''; document.getElementById('exit-lite-phone').value = ''; document.getElementById('exit-lite-notes').value = ''; exitShowPanel('lite'); } else { exitShowPanel('thanks'); document.getElementById('exit-thanks-text').innerHTML = `Sorry we couldn't help this time.
